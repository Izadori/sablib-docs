19const std::vector<double> SavitzkyGolayCoefficients(
20 const unsigned int n, const unsigned int polyorder, const unsigned derive, const double delta
21)
22{
23 unsigned int window = 2 * n + 1;
24 double p = 1;
25
26 assert(window > polyorder && window > derive && polyorder > derive);
27
28 Eigen::VectorXd v = Eigen::VectorXd::LinSpaced(window, (int)-n, n);
29 Eigen::MatrixXd x = Eigen::MatrixXd::Ones(window, polyorder + 1);
30
31 for(unsigned int i = 1; i <= polyorder; i++){
32 x.col(i) = (x.col(i - 1).array() * v.array()).matrix();
33 }
34
35 Eigen::MatrixXd coeff_mat = (x.transpose() * x).inverse() * x.transpose();
36
37 if(derive > 0){
38 for(unsigned int i = 1; i <= derive; i++){
39 p *= i;
40 }
41 p /= std::pow(delta, derive);
42 }
43
44 Eigen::VectorXd coeff = coeff_mat.row(derive) * p;
45 std::vector<double> result(coeff.size());
46
47 Eigen::VectorXd::Map(result.data(), result.size()) = coeff;
48
49 return result;
50}
