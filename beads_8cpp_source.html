24BAfilt(const unsigned int d, const double frequency, const unsigned int length)
25{
26 Eigen::VectorXd a(1), b, b1(2), v(3), v2(2);
27 double omega_c = 2 * M_PI * frequency;
28 double t;
29
30 b1 << 1, -1;
31 v << -1, 2, -1;
32
33 for(unsigned int i = 1; i < d; i++){
34 b1 = Convolve(b1, v).eval();
35 }
36
37 v2 << -1, 1;
38 b = Convolve(b1, v2);
39
40 v << 1, 2, 1;
41 a << 1;
42
43 for(unsigned int i = 0; i < d; i++){
44 a = Convolve(a, v).eval();
45 }
46
47 t = std::pow((1 - std::cos(omega_c)) / (1 + std::cos(omega_c)), d);
48 a = (b + t * a).eval();

50 Eigen::MatrixXd xa(a.size(), length);
51 Eigen::MatrixXd xb(b.size(), length);
52
53 for(unsigned int i = 0; i < length; i++) {
54 xa.block(0, i, a.size(), 1) = a;
55 xb.block(0, i, b.size(), 1) = b;
56 }
57
58 Eigen::VectorXi dr = Eigen::VectorXi::LinSpaced(2 * d + 1, -d, d);
59 Eigen::SparseMatrix<double> A = Spdiags(xa, dr, length, length);
60 Eigen::SparseMatrix<double> B = Spdiags(xb, dr, length, length);
61

70const std::tuple< std::vector<double>, std::vector<double> >
71BaselineBeads(
72 const std::vector<double> & y, const unsigned int s, const double frequency, const double r,
73 const double lambda0, const double lambda1, const double lambda2,
74 const unsigned int loop, const double eps, const BeadsPenalty penalty
75)
76{
77 const double eps0 = 1e-6;
78 const double eps1 = 1e-6;
79
80 Eigen::VectorXd yy = Eigen::VectorXd::Map(y.data(), y.size());
81
82 std::function<double(const double)> phi, wfun;
83
84 switch(penalty) {
85 case BeadsPenalty::L1_v1:
86 phi = [&](const double xx) {
87 double abs_x = std::fabs(xx);
88 return std::sqrt(abs_x * abs_x + eps1);
89 };
90 wfun = [&](const double xx) {
91 return 1 / phi(xx);
92 };
93 break;
94 case BeadsPenalty::L1_v2:
95 phi = [&](const double xx) {
96 double abs_x = std::fabs(xx);
97 return abs_x - eps1 * std::log(abs_x + eps1);
98 };
99 wfun = [&](const double xx) {
100 return 1 / (std::fabs(xx) + eps1);
101 };
102 break;
103 default:
104 throw std::invalid_argument("invalid penalty function type.");
105 }
106
107 auto theta = [&](const double xx) {
108 if(xx > eps0) {
109 return xx;
110 }
111 else if(xx < -eps0) {
112 return -r * xx;
113 }
114 else {
115 return (1 + r) * xx * xx / (4 * eps0) + (1 - r) * xx / 2 + eps0 * (1 + r) / 4;
116 }
117 };
118
119 int length = yy.size();
120 auto [ A, B ] = BAfilt(s, frequency, length);
121
122 Eigen::SparseMatrix<double> I, D1, D2;
123
124 I.resize(length, length);
125 I.setIdentity();
126 D1 = Diff(I);
127 D2 = Diff(I, 2);
128
129 Eigen::SparseLU< Eigen::SparseMatrix<double> > solverA, solverQ;
130
131 solverA.compute(A);
132
133 Eigen::SparseMatrix<double> BTB = B.transpose() * B;
134 Eigen::VectorXd b = Eigen::VectorXd::Constant(length, (1 - r) / 2);
135 Eigen::VectorXd d = BTB * (solverA.solve(yy)) - lambda0 * A.transpose() * b;
136 Eigen::VectorXd w1 = Eigen::VectorXd::Constant(length - 1, lambda1);
137 Eigen::VectorXd w2 = Eigen::VectorXd::Constant(length - 2, lambda2);
138
139 Eigen::VectorXd x = yy;
140
141 double prev_c = 0.5
142 * (B * solverA.solve(x)).unaryExpr([](const double xx){ return xx * xx; }).sum()
143 + lambda0 * x.unaryExpr(theta).sum()
144 + lambda1 * (D1 * x).unaryExpr(phi).sum()
145 + lambda2 * (D2 * x).unaryExpr(phi).sum();
146
147 for(unsigned int i = 0; i < loop; i++) {
148 Eigen::SparseMatrix<double> L1, L2, G, M;
149
150 L1 = (w1.array() * (D1 * x).unaryExpr(wfun).array()).matrix().asDiagonal();
151 L2 = (w2.array() * (D2 * x).unaryExpr(wfun).array()).matrix().asDiagonal();
152
153 G = x.unaryExpr([&](const double xx) {
154 double z = (-eps0 <= xx && xx <= eps0) ? eps0 : std::fabs(xx);
155 return (1 + r) / 4 / z;
156 }).asDiagonal();
157
158 M = 2 * lambda0 * G + D1.transpose() * L1 * D1 + D2.transpose() * L2 * D2;
159 solverQ.compute(BTB + A.transpose() * M * A);
160
161 x = A * solverQ.solve(d);
162
163 Eigen::VectorXd a = yy - x;
164 double c = 0.5
165 * (B * solverA.solve(a)).unaryExpr([](const double xx){ return xx * xx; }).sum()
166 + lambda0 * x.unaryExpr(theta).sum()
167 + lambda1 * (D1 * x).unaryExpr(phi).sum()
168 + lambda2 * (D2 * x).unaryExpr(phi).sum();
169
170 if(std::fabs((prev_c - c) / c) < eps) {
171 break;
172 }
173
174 prev_c = c;
175 }
176
177 Eigen::VectorXd f = yy - x;
178 f = (f - B * solverA.solve(f)).eval();
179
180 std::vector<double> f_result(f.size()), x_result(x.size());
181
182 Eigen::VectorXd::Map(f_result.data(), f.size()) = f;
183 Eigen::VectorXd::Map(x_result.data(), x.size()) = x;
184
185 return std::tuple< std::vector<double>, std::vector<double> >(f_result, x_result);
186}
