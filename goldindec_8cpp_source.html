30 }
31
32 if(peak_ratio <= 0 || 1 < peak_ratio) {
33 throw std::invalid_argument("BaselineGoldindec(): peak_ratio must be between 0 and 1.");
34 }
35
36 if(alpha <= 0 || 1 < alpha) {
37 throw std::invalid_argument("BaselineGoldindec(): alpha must be between 0 and 1.");
38 }
39
40 if(loop == 0) {
41 throw std::invalid_argument("BaselineGoldindec(): loop is zero.");
42 }
43
44 if(eps <= 0) {
45 throw std::invalid_argument("BaselineGoldindec(): non-positive eps value is given.");
46 }
47
48 Eigen::VectorXd yy = Eigen::VectorXd::Map(y.data(), y.size());
49 Eigen::VectorXd x = Eigen::VectorXd::LinSpaced(y.size(), 0, 1);
50 Eigen::MatrixXd V = Vandermonde(x, polyorder);
51 Eigen::LDLT<Eigen::MatrixXd> ldltV = (V.transpose() * V).ldlt();
52
53 double a = 0, b = 1;
54 double rud = 0.7679
55 + 11.2358 * peak_ratio
56 - 39.7064 * peak_ratio * peak_ratio
57 + 92.3583 * peak_ratio * peak_ratio * peak_ratio; // See Liu's paper.
58 double s = a + 0.618 * (b - a), s_old = s;
59
60 auto derivative = [&](const double xx) {
61 if(xx < s) {
62 return 2 * xx;
63 }
64 else {
65 return -s * s * s / (2 * xx * xx);
66 }
67 };
68
69 Eigen::VectorXd coeff = ldltV.solve(V.transpose() * yy);
70 Eigen::VectorXd z = PolyVal(coeff, V);
71
72 for(unsigned int i = 0; i < loop; i++) {
73 // LEGEND algorithm
74 for(unsigned int j = 0; j < loop_legend; j++) {
75 Eigen::VectorXd z_old = z;
76
77 Eigen::VectorXd e = yy - z_old;
78 Eigen::VectorXd d = -e + alpha * e.unaryExpr(derivative);
79 coeff = ldltV.solve(V.transpose() * (yy + d));
80 z = PolyVal(coeff, V);
81
82 if((z - z_old).norm() / z_old.norm() < eps_legend) {
83 break;
84 }
85 }
86
87 double up_down_ratio = (double)(yy.array() >= z.array()).count() / (double)(yy.array() < z.array()).count();
88 double diff = up_down_ratio - rud;
89
90 if(diff > eps) {
91 a = s;
92 }
93 else if(diff < -eps) {
94 b = s;
95 }
96 else {
97 break;
98 }
99
100 s = a + 0.618 * (b - a);
101 // std::cerr << "up_down_ratio = " << up_down_ratio << ", s = " << s << std::endl;
102
103 // Based on experimentation, it seems better to set a threshold for s as well.
104 if(std::fabs(s - s_old) < eps_s) {
105 break;
106 }
107
108 s_old = s;
109 }
110
111 std::vector<double> result(z.size());
112 Eigen::VectorXd::Map(result.data(), result.size()) = z;
113
114 return result;
115}
