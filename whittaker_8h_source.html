54template <typename Derived>
55const typename Derived::PlainObject
56Whittaker(
57 const Eigen::MatrixBase<Derived> & y,
58 const Eigen::MatrixBase<Derived> & w,
59 const Eigen::SparseMatrix<typename Derived::PlainObject::Scalar> & lambdaDTD
60)
61{
62 // Although parameters are received as MatrixBase<Derived>, only vector classes are allowed.
63 // Others will be rejected at compile time.
64 static_assert(Derived::IsVectorAtCompileTime, "Error: y and w are not vector.");
65
66 using Scalar = typename Derived::PlainObject::Scalar;
67
68 Eigen::VectorXd z;
69 Eigen::SparseMatrix<typename Derived::PlainObject::Scalar> W;
70 Eigen::SimplicialCholesky< Eigen::SparseMatrix<Scalar> > solver;
71
72 W = w.asDiagonal();
73 solver.compute(W + lambdaDTD);
74 z = solver.solve(W * y);
75
76 return z;
77}
